\documentclass{standalone}
\usepackage{tikz}
\usepackage{amsmath}

\begin{document}
\begin{tikzpicture}
  % Axes
  \draw[->] (-4.5,0) -- (4.5,0) node[right] {$x$};
  \draw[->] (0,-5.5) -- (0,2.5) node[above] {$y$};
  
  % X-axis labels
  \foreach \x/\xtext in {-4/{-\pi}, -2/{-\frac{\pi}{2}}, 2/{\frac{\pi}{2}}, 4/{\pi}}
    \draw (\x,0.1) -- (\x,-0.1) node[below] {$\xtext$};

  % Y-axis labels
  \foreach \y/\ytext in {-4/{-4}, -3/{-3}, -2/{-2}, -1/{-1}, 1/{1}, 2/{2}}
    \draw (0.1,\y) -- (-0.1,\y) node[left] {$\ytext$};
  
  % Function y = -4 * sin(2π * x / 4 - π/2) + 1
  \draw[domain=-4:4, smooth, variable=\x, blue, thick] plot ({\x}, {-4*sin(360*\x/4 - 90) + 1});
  \node[blue] at (4, 1) {$y = -4 \sin\left(\frac{2\pi x}{4} - \frac{\pi}{2}\right) + 1$};
\end{tikzpicture}
\end{document}
